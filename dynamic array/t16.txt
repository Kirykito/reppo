int find_max(int_vector& v) {
    int n;
    std::cin >> n;
    
    for (int i = 0; i < n; i++) {
        int val;
        std::cin >> val;
        v.push_back(val); 
    }

    int max_val = v.at(0); 
    for (int i = 1; i < v.size(); i++) {
        if (v.at(i) > max_val) {
            max_val = v.at(i);
        }
    }
    return max_val;
}
