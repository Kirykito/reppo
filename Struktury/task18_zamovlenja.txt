#include <stdio.h>
#include <math.h> 

struct Point {
    double x;
    double y;
};

struct Segment {
    struct Point start;
    struct Point end;
};

int main() {
    struct Segment seg;

    printf("Start point - x: ");
    scanf("%lf", &seg.start.x);
    printf("Start point - y: ");
    scanf("%lf", &seg.start.y);
    printf("End point - x: ");
    scanf("%lf", &seg.end.x);
    printf("End point - y: ");
    scanf("%lf", &seg.end.y);
    double dx = seg.end.x - seg.start.x;
    double dy = seg.end.y - seg.start.y;
    double length = sqrt(dx * dx + dy * dy);
    printf("\nSegment length: %.2f\n", length);

    return 0;
}
