#include <stdio.h>
#include <math.h> 

struct Point {
    double x;
    double y;
};

int main() {
    
    struct Point p1, p2;
    printf("Point 1 - Enter x: ");
    scanf("%lf", &p1.x);
    printf("Point 1 - Enter y: ");
    scanf("%lf", &p1.y);
    printf("Point 2 - Enter x: ");
    scanf("%lf", &p2.x);
    printf("Point 2 - Enter y: ");
    scanf("%lf", &p2.y);
    double dx = p2.x - p1.x;
    double dy = p2.y - p1.y;
    double distance = sqrt(dx * dx + dy * dy);
    printf("Distance: %.2f\n", distance);

    return 0;
}
