#include <stdio.h>

struct Student {
    char name[50];
    int grade;
};

int main() {
    struct Student students[3];

    for (int i = 0; i < 3; i++) {
        printf("Student %d - name: ", i + 1);
        scanf(" %[^\n]", students[i].name);
        
        printf("Student %d - grade: ", i + 1);
        scanf("%d", &students[i].grade);
    }

    printf("\nResults:\n");
    for (int i = 0; i < 3; i++) {
        printf("%d. %s - %d\n", i + 1, students[i].name, students[i].grade);
    }

    return 0;
}
