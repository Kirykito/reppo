#include <stdio.h>

struct Rectangle {
    double width;
    double height;
};

int main() {
    
    struct Rectangle rect;
    printf("Enter width: ");
    scanf("%lf", &rect.width);
    printf("Enter height: ");
    scanf("%lf", &rect.height);
    double area = rect.width * rect.height;
    double perimeter = 2 * (rect.width + rect.height);
    printf("Area: %.2f\n", area);
    printf("Perimeter: %.2f\n", perimeter);

    return 0;
}
